Business Case 1: Traffic Signal Control

Using Queue data structure

Implementation for Business Case 1



 
#include <iostream>
#include <queue>
#include <string>

using namespace std;

int main() {
    queue<string> lane1;
    queue<string> lane2;

    // Predefined vehicles for Lane 1
    lane1.push("Car A");
    lane1.push("Car B");
    lane1.push("Car C");

    // Predefined vehicles for Lane 2
    lane2.push("Truck X");
    lane2.push("Truck Y");
    lane2.push("Truck Z");

    const int greenTimePerVehicle = 2;

    // Process Lane 1
    if (!lane1.empty()) {
        int greenTimeLane1 = lane1.size() * greenTimePerVehicle;
        cout << "Green signal for Lane 1 (Duration: " << greenTimeLane1 << " seconds)" << endl;
        while (!lane1.empty()) {
            cout << lane1.front() << " is passing through Lane 1." << endl;
            lane1.pop();
        }
    }

    // Process Lane 2
    if (!lane2.empty()) {
        int greenTimeLane2 = lane2.size() * greenTimePerVehicle;
        cout << "Green signal for Lane 2 (Duration: " << greenTimeLane2 << " seconds)" << endl;
        while (!lane2.empty()) {
            cout << lane2.front() << " is passing through Lane 2." << endl;
            lane2.pop();
        }
    }

    cout << "All vehicles have passed the traffic signal." << endl;

    return 0;
}

Using travelling salesman problem

            #include <iostream>

            #include <vector>

            #include <limits>

            using namespace std;


            const int INF = 1e9;


            int findNearest(int current, const vector<vector<int>>& graph, vector<bool>& visited) {

                int n = graph.size();

                int nearest = -1;

                int minDistance = INF;


                for (int i = 0; i < n; ++i) {

                    if (!visited[i] && graph[current][i] < minDistance) {

                        minDistance = graph[current][i];

                        nearest = i;

                    }

                }


                return nearest;

            }


            void greedyTSP(const vector<vector<int>>& graph) {

                int n = graph.size();

                vector<bool> visited(n, false);

                visited[0] = true;

                int current = 0;

                int totalDistance = 0;


                cout << "Route: 0 ";


                for (int i = 1; i < n; ++i) {

                    int next = findNearest(current, graph, visited);

                    if (next == -1) break;


                    totalDistance += graph[current][next];

                    current = next;

                    visited[current] = true;

                    cout << "-> " << current << " ";

                }


                totalDistance += graph[current][0];

                cout << "-> 0" << endl;


                cout << "Total Distance: " << totalDistance << " units" << endl;

            }


            int main() {

                int n;

                cout << "Enter the number of nodes: ";

                cin >> n;


                vector<vector<int>> graph(n, vector<int>(n));

                cout << "Enter the adjacency matrix (distance between nodes):" << endl;

                for (int i = 0; i < n; ++i) {

                    for (int j = 0; j < n; ++j) {

                        cin >> graph[i][j];

                    }

                }


                greedyTSP(graph);


                return 0;

            }
